kubectl top node/pod  [name| -l label        #查看节点/pod占用的CPU和内存

# 2).设置kubectl与哪个kubernetes集群进行通信并修改配置信息
cat ~/.kube/config                          #查看k8s配置信息
kubectl config view                         #显示合并的kubeconfig配置
kubectl config get-contexts                 #显示上下文列表
kubectl config current-context              #展示当前所处的上下文
kubectl config use-context my-cluster-name #设置默认的的上下文为my-cluster-name
kubectl config set-context gce --user=cluster-admin --namespace=foo \
  && kubectl config use-context gce             # 使用特定的用户名和名字空间设置上下文

# 1).clusterrole相关
kubectl create clusterrole --help       #查看创建clusterrole的命令帮助
#创建clusterrole操作，verb表示动作，比如create resource表示资源比如daemonset
kubectl create clusterrole NAME --verb=verb --resource=resource.group [--resource-name=resourcename] \
  [--dry-run=server|client|none] [options]    
#创建clusterrole空跑并生成yaml清单
kubectl create clusterrole deployment-clusterrole --verb=create \
  --resource=deployments,daemonsets,statefulsets --dry-run=client -o yaml > 1.clusterrole.yaml  

# 2).servicecount(sa)相关
kubectl create ns app-team1
kubectl -n app-team1 create serviceaccount cicd-token

# 3).binding相关
kubectl create rolebinding --help      #查看创建rolebinding的命令帮助
#创建rolebinding的命令
kubectl create rolebinding NAME --clusterrole=NAME|--role=NAME [--user=username] [--group=groupname] \
  [--serviceaccount=namespace:serviceaccountname] [--dry-run=server|client|none] [options]
#创建具体的rolebinding绑定的实例
kubectl create rolebinding deployment-rolebinding --clusterrole=deployment-clusterrole --serviceaccount=app-team1:cicd-token \
  -n=app-team1 --dry-run=client -o yaml > rolebinding.yaml

# 1). cordon维护
kubect cordon/uncordon node01                                       #设置为维护状态不可调度和恢复
# 2). taint污点
kubectl taint nodes master1 kubernetes.io/role1=master1:NoSchedule  #给master1节点增加污点
kubectl describe nodes master1                                      #查看master1节点是否有污点
kubectl taint nodes master1 kubernetes.io/role1=master1:NoSchedule- #给master1节点删除污点
# 3). drain 驱逐
kubectl drain -h                                                    #drain为驱逐
--delete-local-data=false: Continue even if there are pods using emptyDir (local data that will be deleted when  the node is drained).
--ignore-daemonsets=false: Ignore DaemonSet-managed pods.
--force=false: Continue even if there are pods not managed by a ReplicationController, ReplicaSet, Job, DaemonSet  or StatefulSet.
kubectl drain node01 --force --delete-local-data --ignore-daemonsets #drain会先做condon，所以不需要单独做cordon了

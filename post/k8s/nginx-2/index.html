apiVersion: v1
kind: Namespace

  name: soft
---
apiVersion: apps/v1
kind: Deployment
metadata:
  namespace: soft
  name: nginx-deploy
spec:
  selector:
    matchLabels:
      app: nginx
  replicas: 2
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: 10.225.68.178/soft/nginx:1.19
        ports:
        - containerPort: 80
